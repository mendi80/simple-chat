*sql create table

IF OBJECT_ID(users) IS NOT NULL DROP TABLE users
IF OBJECT_ID(posts) IS NOT NULL DROP TABLE posts

CREATE TABLE users (
  user_id		INT				NOT NULL PRIMARY KEY AUTO_INCREMENT,
  created 		DATETIME 		NOT NULL,
  nickname		VARCHAR(16) 	NOT NULL,
  secret		VARCHAR(16) 	NOT NULL
 ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE posts (
  post_id 		INT				NOT NULL PRIMARY KEY AUTO_INCREMENT,
  ppost_id 		INT				NOT NULL,
  rpost_id 		INT				NOT NULL,
  user_id 		INT				NOT NULL,
  created 		DATETIME 		NOT NULL,
  nickname		VARCHAR(16) 	NOT NULL,
  secret		VARCHAR(16) 	NOT NULL,
  title 		VARCHAR(200)	NOT NULL,
  content		VARCHAR(2000)	NOT NULL,
  blockedit		INT				NOT NULL,
  blockreply	INT				NOT NULL,
  CONSTRAINT 	FK_USERID FOREIGN KEY (user_id) REFERENCES users(user_id),
  CONSTRAINT 	FK_PPOSTID FOREIGN KEY (ppost_id) REFERENCES posts(post_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE replies (
  reply_id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
  created_datetime datetime NOT NULL,
  user_id int,
  nickname varchar(16) NOT NULL,
  title text,
  content text,
  FOREIGN KEY (user_id) REFERENCES Users(user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
